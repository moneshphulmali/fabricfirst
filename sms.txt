function testSendSMS($mobile, $message) {
    $api_key = 'oaCurOd1Ki0SAIHtPwx5gs9fycFXZQ6JpLB8Yz2W7D3vqebElMhcFHNM89nPRTIDo3fjQJig6Y5skdmr';
    
    // Clean mobile number
    $mobile = preg_replace('/\D/', '', $mobile);
    
    // Validate
    if (strlen($mobile) != 10) {
        return ['error' => 'Invalid mobile number (must be 10 digits)'];
    }
    
    $url = "https://www.fast2sms.com/dev/bulkV2";
    
    // Test different settings
    $test_configs = [
        ['route' => 'p', 'sender' => 'FSTSMS', 'name' => 'Promotional (p) with FSTSMS'],
        ['route' => 'p', 'sender' => 'TXTIND', 'name' => 'Promotional (p) with TXTIND'],
        ['route' => 'v3', 'sender' => 'FSTSMS', 'name' => 'Route v3 with FSTSMS'],
        ['route' => 'dlt', 'sender' => 'FSTSMS', 'name' => 'DLT Route'],
    ];
    
    $all_results = [];
    
    foreach ($test_configs as $config) {
        echo "<div style='border:1px solid #ccc; padding:15px; margin:10px 0; border-radius:8px;'>";
        echo "<h3 style='color:#1976d2;'>Testing: {$config['name']}</h3>";
        
        $fields = [
            "sender_id" => $config['sender'],
            "message" => $message,
            "language" => "english",
            "route" => $config['route'],
            "numbers" => $mobile,
            "flash" => 0
        ];
        
        echo "<p><strong>Settings:</strong> Sender: {$config['sender']}, Route: {$config['route']}</p>";
        
        $headers = [
            'authorization: ' . $api_key,
            'Content-Type: application/json',
            'cache-control: no-cache'
        ];
        
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_POST, true);
        curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, 0);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($fields));
        curl_setopt($ch, CURLOPT_TIMEOUT, 10);
        
        $result = curl_exec($ch);
        $curl_error = curl_error($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        
        echo "<p><strong>HTTP Status:</strong> $http_code</p>";
        
        if ($curl_error) {
            echo "<p style='color:red;'><strong>cURL Error:</strong> $curl_error</p>";
        }
        
        $response = json_decode($result, true);
        
        echo "<p><strong>API Response:</strong></p>";
        echo "<pre style='background:#f5f5f5; padding:10px; border-radius:5px;'>";
        print_r($response);
        echo "</pre>";
        
        if (isset($response['return']) && $response['return'] == true) {
            echo "<p style='color:green; font-weight:bold;'>✅ SMS SENT SUCCESSFULLY!</p>";
        } else {
            echo "<p style='color:red; font-weight:bold;'>❌ SMS FAILED</p>";
            if (isset($response['message'])) {
                echo "<p><strong>Error:</strong> " . $response['message'] . "</p>";
            }
        }
        
        echo "</div><hr>";
        
        $all_results[] = [
            'config' => $config['name'],
            'success' => isset($response['return']) && $response['return'] == true,
            'response' => $response
        ];
    }
    
    return $all_results;
}
