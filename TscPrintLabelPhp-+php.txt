<?php
// file: print_label.php

// Sample dynamic values (replace with your database values)
$customer = "Rahul Sharma";
$order_no = "ORD12345";
$date = date("d-m-Y");
$product = "Maxx Spray";
$service = "5 Litres";
$qty = 3;

// TSPL Label Format
tspl = "
SIZE 60 mm,40 mm
GAP 3 mm,0
DIRECTION 1
CLS

TEXT 20,20,\"3\",0,1,1,\"Customer: $customer\"
TEXT 20,60,\"3\",0,1,1,\"Order No: $order_no\"
TEXT 20,100,\"3\",0,1,1,\"Date: $date\"

TEXT 20,150,\"3\",0,1,1,\"Product: $product\"
TEXT 20,190,\"3\",0,1,1,\"Service: $service\"
TEXT 20,230,\"3\",0,1,1,\"Total Qty: $qty\"

PRINT 1
";

// Send TSPL to NodeJS Print Server
$ch = curl_init("http://localhost:3000/print");
curl_setopt($ch, CURLOPT_POST, 1);
curl_setopt($ch, CURLOPT_POSTFIELDS, $tspl);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$response = curl_exec($ch);
curl_close($ch);

echo $response;
?>
